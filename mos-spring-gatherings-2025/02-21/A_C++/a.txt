int rec_sol(ll cv, ll pos, bool minus) {
    if (pos == -1) {
        if (cv == 0) return 0;
        return 1e9;
    }
    // auto it = cache[pos].find(cv);
    // if (it != cache[pos].end()) return it->second;
    int ans = rec_sol(cv, pos - 1, minus);
    ans = std::min(ans, rec_sol(cv + fvs[pos], pos - 1, true) + 1);
    if (!minus) ans = std::min(ans, rec_sol(cv - fvs[pos], pos - 1, false) + 1);
    // cache[pos].emplace(cv, ans);
    return ans;
}

bool used[100];

int stupid_get_ans(ll n) {
    return rec_sol(n, upper_bound(fvs.begin(), fvs.end(), n) - fvs.begin(), false);
}

int get_ans(ll n) {
    int ans = 0;
    for (int i = 0; i < fvs.size(); ++i) used[i] = false;
    for (ll tn = n, i = fvs.size() - 1; i >= 0; --i) if (fvs[i] <= tn) used[i] = true, tn -= fvs[i], ++ans;
    int ncnt = 0;
    for (int i = fvs.size() - 1; i >= 0; --i) {
        if (used[i]) {
            n -= fvs[i];
            ++ncnt;
            continue;
        }
        ll tn = fvs[i] - n;
        int ccnt = ncnt + 1;
        for (int j = i - 1; j >= 0; --j) if (fvs[j] <= tn) tn -= fvs[j], ++ccnt;
        ans = std::min(ans, ccnt);
    }
    return ans;
}
