constexpr int maxn = 1e5 + 10;

int d;

int _rec_sol(int pos, int us, int lcnt, int sz, int arr[maxn]) {
    if (pos == sz) {
        if (us == d && lcnt == 0) return 0;
        return 1e9;
    }
    if (us > d) return 1e9;
    int ans = std::max(lcnt + arr[pos], _rec_sol(pos + 1, us + 1, 0, sz, arr));
    ans = std::min(ans, _rec_sol(pos + 1, us, lcnt + arr[pos], sz, arr));
    return ans;
}

int rec_sol(int sz, int arr[maxn]) {
    if (sz < d) return 1e9;
    return _rec_sol(0, 0, 0, sz, arr);
}
