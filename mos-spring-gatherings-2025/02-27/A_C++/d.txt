constexpr int maxn = 2e5 + 10;

int n;
int p[maxn];
vector<int> tps[maxn];
int used[maxn];
int timer = 0;

bool check_way(int v, int to) {
    ++timer;
    while (v != to && used[v] != timer) {
        used[v] = timer;
        v = p[v];
    }
    return v == to;
}

void add_way(int v, int to) {
    int t = 0;
    while (v != to) tps[v].push_back(t++), v = p[v];
}
