#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2")

#include <immintrin.h>
#include <iostream>

using namespace std;

constexpr int maxn = 250010, maxt = 10010;

alignas(32) int ds[maxn], as[maxn], bs[maxn], mbs[maxn], ms[maxn];
alignas(32) int lords[maxt];
int n, t;
alignas(32) int ftbs[maxn];

void solve() {
    cin >> n >> t;
    for (int i = 0; i < n; ++i) cin >> ds[i] >> as[i] >> bs[i] >> ms[i], mbs[i] = -bs[i], --ms[i];
    for (int i = 0; i < t; ++i) cin >> lords[i];
    __m256i zr = _mm256_setzero_si256();
    int i = 0;
    for (; i + 8 <= n; i += 8) {
        __m256i cds = _mm256_load_si256((__m256i *) (ds + i));
        __m256i cas = _mm256_load_si256((__m256i *) (as + i));
        __m256i cmbs = _mm256_load_si256((__m256i *) (mbs + i));
        __m256i cms = _mm256_load_si256((__m256i *) (ms + i));
        __m256i cvs = _mm256_setzero_si256();
        __m256i cans = _mm256_setzero_si256();
        for (int j = 0; j < t; ++j) {
            __m256i cld = _mm256_set1_epi32(lords[j]);
            __m256i plf = _mm256_cmpgt_epi32(cld, cds);
            cvs = _mm256_add_epi32(cvs, _mm256_and_si256(plf, cas));
            cvs = _mm256_add_epi32(cvs, _mm256_andnot_si256(plf, cmbs));
            cvs = _mm256_max_epi32(cvs, zr);
            cans = _mm256_or_si256(cans, _mm256_cmpgt_epi32(cvs, cms));
        }
        _mm256_store_si256((__m256i *) (ftbs + i), cans);
    }
    for (; i < n; ++i) {
        int cnt = 0;
        for (int j = 0; j < t; ++j) {
            if (ds[i] >= lords[j]) cnt -= bs[i];
            else cnt += as[i];
            if (cnt < 0) cnt = 0;
            if (cnt >= ms[i]) ftbs[i] = 1;
        }
    }
    for (i = 0; i < n; ++i) cout << (ftbs[i] ? '1' : '0') << " ";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#endif
    solve();
    return 0;
}
