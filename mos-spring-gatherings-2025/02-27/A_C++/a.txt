using namespace std;

constexpr int maxn = 1010, maxq = 2e5 + 10;

pair<int, int> to[maxn][maxn];
bool cycle[maxn][maxn], conv[maxn][maxn];
char dir[maxn][maxn];
int used[maxn][maxn];
int timer = 0;
int n, q;
vector<pair<int, int> > cur_made_cycle;
vector<pair<int, int> > made_cycled[maxq];

pair<pair<int, int>, bool> conv_dfs(int i, int j) {
    if (!conv[i][j]) return {{i, j}, false};
    if (used[i][j] == timer) {
        if (!cycle[i][j]) cur_made_cycle.emplace_back(i, j);
        cycle[i][j] = true, to[i][j] = {i, j};
    }
    if (cycle[i][j]) return {to[i][j], true};
    used[i][j] = timer;
    if (to[i][j].first == -1) return {to[i][j], false};
    auto [ni,nj] = to[i][j];
    auto [mto, ic] = conv_dfs(ni, nj);
    if (ic && !cycle[i][j]) cur_made_cycle.emplace_back(i, j);
    to[i][j] = mto, cycle[i][j] = ic;
}

deque<pair<int, int> > rbq;

void re_bfs(int qid) {
    ++timer;
    rbq.clear();
    for (auto [i,j]: cur_made_cycle) {
        rbq.emplace_back(i, j);
        used[i][j] = timer;
    }
    while (!rbq.empty()) {
        auto [i,j] = rbq.front();
        rbq.pop_front();
        made_cycled[qid].emplace_back(i, j);
        if (i && conv[i - 1][j] && dir[i - 1][j] == 2 && !cycle[i - 1][j] && used[i - 1][j] != timer)
            rbq.emplace_back(i - 1, j), used[i - 1][j] = timer;
        if (j + 1 < n && conv[i][j + 1] && dir[i][j + 1] == 3 && !cycle[i][j + 1] && used[i][j + 1] != timer)
            rbq.emplace_back(i, j + 1), used[i][j + 1] = timer;
        if (i + 1 < n && conv[i + 1][j] && dir[i + 1][j] == 0 && !cycle[i + 1][j] && used[i + 1][j] != timer)
            rbq.emplace_back(i + 1, j), used[i + 1][j] = timer;
        if (j && conv[i][j - 1] && dir[i][j - 1] == 1 && !cycle[i][j - 1] && used[i][j - 1] != timer)
            rbq.emplace_back(i, j - 1), used[i][j - 1] = timer;
    }
}
