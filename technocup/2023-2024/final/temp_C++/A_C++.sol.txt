#include <iostream>

using namespace std;

int main() {
    int64_t a, b, n, m;
    cin >> a >> b >> n >> m;
    if (m == 2 * n) {
        cout << n << " " << 0;
        return 0;
    }
    if (2 * a == b) {
        cout << "-1";
        return 0;
    }
    if ((m - 2 * n) % (b - 2 * a) != 0) {
        cout << "-1";
        return 0;
    }
    int64_t k2 = (m - 2 * n) / (b - 2 * a);
    if (k2 < 0) {
        cout << "-1";
        return 0;
    }
    int64_t k1 = n - a * k2;
    if (k1 < 0) {
        cout << "-1";
        return 0;
    }
    cout << k1 << " " << k2;
    return 0;
}
