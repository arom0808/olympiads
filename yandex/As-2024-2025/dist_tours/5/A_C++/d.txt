#include <iostream>

using namespace std;
using ll = int64_t;
constexpr int maxn = 20;
int n;
int arr[maxn];
int fbn[maxn + 1];

void main_init() {
    fbn[0] = 1, fbn[1] = 2;
    for (int i = 2; i <= maxn; ++i) fbn[i] = fbn[i - 1] + fbn[i - 2];
}

void solve() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> arr[i], --arr[i];
    int cnum = 0;
    for (int i = 0; i < n; ++i) {
        cnum += fbn[arr[i]];
        ll ans = 0;
        for (int mask = 0; mask < (1 << maxn); ++mask) {
            int csum = 0;
            for (int j = 0; j < maxn; ++j) if ((mask >> j) & 1) csum += fbn[j];
            if (csum == cnum) ++ans;
        }
        cout << ans << "\n";
    }
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#endif
    main_init();
    solve();
    return 0;
}
