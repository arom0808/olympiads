constexpr int maxn = 1 << 21;

int val_in_timer = 0;

int stupid_solve_for_arr(int n) {
    int ans = 0;
    for (int start = 0; start < (1 << n); ++start) {
        for (int end = start + 1; end <= (1 << n); ++end) {
            bool ok = false;
            for (int swap1 = 0; swap1 < (1 << n) && !ok; ++swap1) {
                for (int swap2 = swap1 + 1; swap2 < (1 << n) && !ok; ++swap2) {
                    swap(arr[swap1], arr[swap2]);
                    int xor_val = (1 << n) - 1;
                    for (int i = start; i < end; ++i) xor_val ^= arr[i];
                    if (xor_val == 0) ok = true;
                    swap(arr[swap1], arr[swap2]);
                }
            }
            if (ok) ++ans;
        }
    }
    return ans;
}


int solve_for_arr(int n) {
    int ans = 0;
    for (int start = 0; start < (1 << n); ++start) {
        int arr_xor = (1 << n) - 1;
        ++val_in_timer;
        for (int end = start + 1; end <= (1 << n); ++end) {
            arr_xor ^= arr[end - 1];
            val_in[arr[end - 1]] = val_in_timer;
            if (!arr_xor && (end - start >= 2 || (1 << n) - (end - start) >= 2)) {
                ++ans;
                continue;
            }
            for (int i = start; i < end; ++i) {
                if (val_in[arr_xor ^ arr[i]] != val_in_timer) {
                    ++ans;
                    break;
                }
            }
        }
    }
    return ans;
}
