#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
using ll = int64_t;
using ull = uint64_t;
using ld = long double;

void solve() {
    int n;
    cin >> n;
    vector<pair<int, string> > arr(n);
    for (auto &student: arr) cin >> student.second >> student.first;
    sort(arr.begin(), arr.end());
    vector<string> ans(n);
    for (int i = 0; i * 2 < n; ++i) ans[i * 2] = arr[i].second;
    for (int i = 0; i * 2 + 1 < n; ++i) ans[i * 2 + 1] = arr[n - i - 1].second;
    for (const auto &name: ans) cout << name << " ";
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    int t;
    cin >> t;
    while (t--) solve();
#else
    solve();
#endif
    return 0;
}
