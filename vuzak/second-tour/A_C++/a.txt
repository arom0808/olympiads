#include <iostream>

using namespace std;
using ll = long long;

int maxn = 1000;

int n, m;

void solve() {
    cin >> n >> m;
    ll pf = 0;
    int ans = 0;
    for (int i = 0, val; i < n; ++i) {
        cin >> val;
        pf += val;
        if (pf % m == 0) ++ans;
    }
    cout << ans << "\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#endif
    solve();
    return 0;
}
