#include <iostream>

using namespace std;
using ll = long long;

constexpr int maxn = 5e5 + 10, inf = 2e9, maxlogn = 20;

int start_arr[maxn];
int n;

namespace ST {
    int tree[maxlogn][maxn];

    void init() {
        for (int i = 0; i < n; ++i) tree[0][i] = start_arr[i];
        for (int k = 1; k < maxlogn; ++k)
            for (int i = 0; i + (1 << k) <= n; ++i)
                tree[k][i] = std::min(tree[k - 1][i], tree[k - 1][i + (1 << (k - 1))]);
    }

    int min_val(int a, int b) {
        int k = __lg(b - a);
        return std::min(tree[k][a], tree[k][b - (1 << k)]);
    }
}

bool can(int lev, int x, int len) {
    int min_val = ST::min_val(lev, lev + len);
    return x + min_val + lev >= 0;
}

int get_ans(int lev, int x) {
    int l = 0, r = n - lev + 1;
    while (r - l != 1) {
        int m = (l + r) >> 1;
        if (can(lev, x, m)) l = m;
        else r = m;
    }
    return l;
}

void solve() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> start_arr[i], start_arr[i] = -start_arr[i] - i;
    ST::init();
    int q;
    cin >> q;
    for (int _ = 0; _ < q; ++_) {
        int l, x;
        cin >> l >> x, --l;
        cout << get_ans(l, x) << "\n";
    }
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#endif
    solve();
    return 0;
}
