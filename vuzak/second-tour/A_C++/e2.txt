ll vdist[maxn];

void djikstra(int s) {
    for (int i = 0; i < n; ++i) vdist[i] = inf;
    vdist[s] = 0;
    priority_queue<pair<ll, ll>, vector<pair<ll, ll> >, greater<pair<ll, ll> > > q;
    q.emplace(0, s);
    while (!q.empty()) {
        auto [d, v] = q.top();
        q.pop();
        if (d > vdist[v]) continue;
        for (auto [u, w]: g[v]) {
            if (vdist[u] > vdist[v] + w) {
                vdist[u] = vdist[v] + w;
                q.emplace(vdist[u], u);
            }
        }
    }
}
