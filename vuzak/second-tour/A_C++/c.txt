#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
using ll = long long;

constexpr int maxn = 1e5;
int a[maxn], b[maxn], acnt[maxn * 2], bcnt[maxn * 2];
vector<int> all_vals;
int n;

int zip_val(int val) {
    return lower_bound(all_vals.begin(), all_vals.end(), val) - all_vals.begin();
}

void solve() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> a[i], all_vals.push_back(a[i]);
    for (int i = 0; i < n; ++i) cin >> b[i], all_vals.push_back(b[i]);
    sort(all_vals.begin(), all_vals.end());
    all_vals.erase(unique(all_vals.begin(), all_vals.end()), all_vals.end());
    for (int i = 0; i < n; ++i) a[i] = zip_val(a[i]);
    for (int i = 0; i < n; ++i) b[i] = zip_val(b[i]);
    for (int i = 0; i < n; ++i) ++acnt[a[i]];
    for (int i = 0; i < n; ++i) ++bcnt[b[i]];
    int d1 = -1, d2 = -1;
    for (int i = 0; i < all_vals.size(); ++i) {
        if (acnt[i] == bcnt[i]) continue;
        if (d1 == -1) d1 = i;
        else if (d2 == -1) d2 = i;
        else {
            cout << "-1\n";
            return;
        }
    }
    if (d1 == -1) cout << "0\n";
    else if (d2 == -1 || (acnt[d1] == bcnt[d2] && acnt[d2] == bcnt[d1])) cout << "1\n";
    else cout << "-1\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#endif
    solve();
    return 0;
}
