#include <iostream>
#include <vector>
#include <cstdint>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
    int32_t n, m, k;
    cin >> n >> m >> k;
    vector<int32_t> list(n), target(m);
    for (auto &el: list) cin >> el, --el;
    for (auto &el: target) cin >> el, --el;
    vector<int32_t> counts_list(k), counts_target(k);
    for (int32_t i = 0; i < m; ++i) ++counts_list[list[i]], ++counts_target[target[i]];
    int32_t diff = 0;
    for (int32_t i = 0; i < k; ++i) diff += abs(counts_target[i] - counts_list[i]);
    int32_t res = 0;
    if (diff <= 2) ++res;
    for (uint32_t i = 0; i < n - m; ++i) {
        diff -= abs(counts_target[list[i]] - counts_list[list[i]]);
        diff -= abs(counts_target[list[i + m]] - counts_list[list[i + m]]);
        --counts_list[list[i]];
        ++counts_list[list[i + m]];
        diff += abs(counts_target[list[i]] - counts_list[list[i]]);
        diff += abs(counts_target[list[i + m]] - counts_list[list[i + m]]);
        if (diff <= 2) ++res;
    }
    cout << res;
    return 0;
}
