bool check_bca_exists(const string &str) {
    auto bi = str.find("b"), ai = str.rfind("a");
    if (bi == string::npos || ai == string::npos || bi > ai) return false;
    auto ci = str.find("c", bi, ai - bi);
    return ci != string::npos;
}

bool check_abc_exists(const string &str) {
    auto bi = str.find("a"), ai = str.rfind("c");
    if (bi == string::npos || ai == string::npos || bi > ai) return false;
    auto ci = str.find("b", bi, ai - bi);
    return ci != string::npos;
}

bool check_ab_exists(const string &str) {
    auto ai = str.find("a"), bi = str.rfind("b");
    return !(ai == string::npos || bi == string::npos || ai > bi);
}

bool check_bc_exists(const string &str) {
    auto ai = str.find("b"), bi = str.rfind("c");
    return !(ai == string::npos || bi == string::npos || ai > bi);
}

    ios_base::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
    int32_t n;
    cin >> n;
    vector<string> strs(n);
    for (auto &s: strs) cin >> s;
    set<int32_t> indices;
    for (int32_t i = 0; i < n; ++i) indices.insert(i);
    int32_t bca_cnt = 0, full_cnt = 0, ac_cnt = 0;
    vector<bool> is_a(n), is_b(n), is_c(n), is_bca(n, false);
    vector<int32_t> res;
    for (int32_t i = 0; i < n; ++i) {
        is_a[i] = find(strs[i].begin(), strs[i].end(), 'a') != strs[i].end();
        is_b[i] = find(strs[i].begin(), strs[i].end(), 'b') != strs[i].end();
        is_c[i] = find(strs[i].begin(), strs[i].end(), 'c') != strs[i].end();
        if (is_a[i] && is_c[i]) ++ac_cnt;
        if (!is_a[i]) {
            res.push_back(i);
            indices.erase(i);
        } else {
            if (check_abc_exists(strs[i])) {
                cout << "No";
                return 0;
            }
            if (check_bca_exists(strs[i])) {
                is_bca[i] = true;
                ++bca_cnt;
            }
            if (is_a[i] && is_b[i] && is_c[i]) ++full_cnt;
        }
    }
    if (bca_cnt > 1) {
        cout << "No";
        return 0;
    }
    if (full_cnt >= 3) {
        cout << "No";
        return 0;
    }
    if (full_cnt == 0) {
        for (int32_t i = 0; i < n; ++i) {
            if (indices.contains(i) && is_a[i] && is_c[i]) {
                res.push_back(i);
                indices.erase(i);
            }
        }
    } else if (full_cnt == 1) {
        int32_t full_i = 0;
        for (int32_t i = 0; i < n; ++i) {
            if (is_a[i] && is_b[i] && is_c[i]) {
                full_i = i;
                break;
            }
        }
        if (check_ab_exists(strs[full_i])) {
            for (int32_t i = 0; i < n; ++i) {
                if (is_a[i] && is_c[i]) {
                    res.push_back(i);
                    indices.erase(i);
                }
            }
            res.push_back(full_i);
            indices.erase(full_i);
        } else {
            res.push_back(full_i);
            indices.erase(full_i);
            for (int32_t i = 0; i < n; ++i) {
                if (indices.contains(i) && is_a[i] && is_c[i]) {
                    res.push_back(i);
                    indices.erase(i);
                }
            }
        }
    } else if (full_cnt == 2) {
        if (ac_cnt == 0) {
            for (int32_t i = 0; i < n; ++i) {
                if (is_a[i] && is_b[i] && is_c[i]) {
                    res.push_back(i);
                    indices.erase(i);
                }
            }
        } else {
            for (int32_t i = 0; i < n; ++i) {
                if (is_a[i] && is_b[i] && is_c[i] && !check_ab_exists(strs[i])) {
                    res.push_back(i);
                    indices.erase(i);
                    --full_cnt;
                    break;
                }
            }
            if (full_cnt == 2) {
                cout << "No";
                return 0;
            }
            for (int32_t i = 0; i < n; ++i) {
                if (indices.contains(i) && is_a[i] && is_b[i] && is_c[i] && !check_bc_exists(strs[i])) {
                    res.push_back(i);
                    indices.erase(i);
                    --full_cnt;
                    break;
                }
            }
            if (full_cnt == 1) {
                cout << "No";
                return 0;
            }
        }
    }
    for (auto ind: indices) res.push_back(ind);
    cout << "Yes\n";
    for (auto el: res) cout << el + 1 << " ";
