#include <bits/stdc++.h>

using namespace std;
using ll = long long;

constexpr int maxn = 1e5;
int as[maxn], bs[maxn];
int n;

bool check(ll C) {
    ll min_val = 0, max_val = 0;
    for (int i = 0; i < n; ++i) {
        ll cmin = std::max(C - bs[i], 0ll), cmax = as[i];
        if (cmin > cmax) return false;
        min_val += cmin, max_val += cmax;
    }
    return min_val <= C && C <= max_val;
}

void solve() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> as[i];
    for (int i = 0; i < n; ++i) cin >> bs[i];
    ll l = 0, r = 1e17;
    while (r - l != 1) {
        ll mid = (l + r) / 2;
        if (check(mid))
            l = mid;
        else
            r = mid;
    }
    cout << l << "\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
    int cfmod = 0, t = 1;
#ifdef LOCAL
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    cfmod = 1;
#endif
    if (cfmod) cin >> t;
    while (t--) solve();
    return 0;
}
