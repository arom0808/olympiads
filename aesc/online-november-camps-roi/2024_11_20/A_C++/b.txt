#include <iostream>
#include <vector>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

using namespace std;
using namespace __gnu_pbds;
using ll = int64_t;
template<typename T>
using ordered_multiset = tree<T, null_type, less_equal<T>, rb_tree_tag, tree_order_statistics_node_update>;

constexpr int maxn = 1e5;

template<typename T>
T mceil(T a, T b) {
    return (a + b - 1) / b;
}

int corp_arr[maxn], live_arr[maxn], arr1[maxn * 2], arr2[maxn * 2];
int n;
int otd_corp_pos[maxn], otd_live_pos[maxn];

int cnt_bigger(const ordered_multiset<int> &co, int val) {
    return static_cast<int>(co.size() - co.order_of_key(val + 1));
}

int cnt_lower(const ordered_multiset<int> &co, int val) {
    return static_cast<int>(co.order_of_key(val));
}

void solve() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> corp_arr[i], --corp_arr[i], otd_corp_pos[corp_arr[i]] = i;
    for (int i = 0; i < n; ++i) cin >> live_arr[i], --live_arr[i], otd_live_pos[live_arr[i]] = i;
    for (int i = 0; i < n; ++i) arr1[i] = otd_live_pos[corp_arr[i]], arr2[i] = otd_corp_pos[live_arr[i]];
    for (int i = 0; i < n; ++i) arr1[i + n] = arr1[i], arr2[i + n] = arr2[i];
    ordered_multiset<int> o1, o2;
    ll res1 = 0, res2 = 0;
    for (int i = 0; i < n; ++i) {
        res1 += cnt_bigger(o1, arr1[i]);
        res2 += cnt_bigger(o2, arr2[i]);
        o1.insert(arr1[i]);
        o2.insert(arr2[i]);
    }
    ll ans = std::min(res1, res2);
    for (int ast = 1; ast < n; ++ast) {
        res1 -= cnt_lower(o1, arr1[ast - 1]);
        res2 -= cnt_lower(o2, arr2[ast - 1]);
        o1.erase(o1.find_by_order(o1.order_of_key(arr1[ast - 1])));
        o2.erase(o2.find_by_order(o2.order_of_key(arr2[ast - 1])));
        res1 += cnt_bigger(o1, arr1[ast + n - 1]);
        res2 += cnt_bigger(o2, arr2[ast + n - 1]);
        o1.insert(arr1[ast + n - 1]);
        o2.insert(arr2[ast + n - 1]);
        ans = std::min(ans, std::min(res1, res2));
    }
    cout << ans << "\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif
    solve();
    return 0;
}
