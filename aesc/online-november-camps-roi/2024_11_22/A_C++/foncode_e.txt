#include <iostream>

using namespace std;
using ll = int64_t;

int n, m, k;
constexpr int maxm = 100, maxk = 500;
int serv_cnt[maxm];

ll dp[maxm + 1][maxk + 1];

ll calc_seg_cost(ll len) {
    return len * (len + 1) / 2;
}

ll calc_server_min(ll cqrs, ll ck) {
    ll mndist = cqrs / (ck + 1);
    ll x = ck * (mndist + 1) + mndist + 1 - cqrs;
    return x * calc_seg_cost(mndist) + (ck + 1 - x) * calc_seg_cost(mndist + 1);
}

void solve() {
    cin >> n >> m >> k;
    for (int i = 0, val; i < n; ++i) {
        cin >> val;
        ++serv_cnt[val - 1];
    }
    for (int i = 0; i <= m; ++i) for (int j = 0; j <= k; ++j) dp[i][j] = 1e18;
    for (int i = 0; i <= k; ++i) dp[0][i] = 0;
    for (int i = 1; i <= m; ++i) {
        for (int j = 0; j <= k; ++j) {
            for (int cc = 0; cc <= j; ++cc) {
                dp[i][j] = std::min(dp[i][j], dp[i - 1][j - cc] + calc_server_min(serv_cnt[i - 1], cc));
            }
        }
    }
    cout << dp[m][k];
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif
    solve();
    return 0;
}
