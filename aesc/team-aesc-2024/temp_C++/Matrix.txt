const int32_t maxn = 1 << 9;
int32_t k, n;

int32_t table[maxn][maxn];
vector<int32_t> g[maxn];
int32_t rm[maxn];
int32_t used[maxn];
int32_t timer;

bool dfs(int32_t v) {
    if (used[v] == timer) return false;
    used[v] = timer;
    for (auto u: g[v]) {
        if (rm[u] == -1) {
            rm[u] = v;
            return true;
        }
    }
    for (auto u: g[v]) {
        if (dfs(rm[u])) {
            rm[u] = v;
            return true;
        }
    }
    return false;
}
