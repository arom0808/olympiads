#include <iostream>
#include <algorithm>
#include <vector>
#include <iomanip>

using namespace std;
using ll = long long;
using ld = long double;

constexpr int maxn = 20;
constexpr ld eps = 1e-9;

int n;
bool table[maxn][maxn];

pair<ld, ld> get_center_by_pixel(ll x, ll y) {
    return {x + 0.5l, y + 0.5l};
}

int cnt_around(int x, int y) {
    int ans = 0;
    if (x) ans += table[x - 1][y];
    if (x + 1 < n) ans += table[x + 1][y];
    if (y) ans += table[x][y - 1];
    if (y + 1 < n) ans += table[x][y + 1];
    return ans;
}

bool check_with_center(ld xc, ld yc, ld rad) {
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (!table[i][j]) continue;
            auto [x,y] = get_center_by_pixel(i, j);
            if ((x - xc) * (x - xc) + (y - yc) * (y - yc) >= rad * rad) return false;
        }
    }
    return true;
}

void solve() {
    cin >> n;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            char c;
            cin >> c;
            table[i][j] = c == '#';
        }
    }
    vector<pair<ld, ld> > border_points;
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            if (table[i][j])
                border_points.push_back(get_center_by_pixel(i, j));
    pair<ld, ld> center;
    for (auto &[x,y]: border_points) center.first += x, center.second += y;
    center.first /= border_points.size();
    center.second /= border_points.size();
    ld l = 0, r = 50;
    while (r - l > eps) {
        ld mid = (l + r) / 2;
        if (check_with_center(center.first, center.second, mid)) r = mid;
        else l = mid;
    }
    cout << fixed << setprecision(10);
    cout << center.second << " " << center.first << " " << r << "\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#elifdef TEST
    freopen("input.txt", "r", stdin);
#endif
    solve();
    return 0;
}
