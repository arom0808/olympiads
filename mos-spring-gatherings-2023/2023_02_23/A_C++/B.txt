#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
// using __int128_t = long long;

constexpr __int128_t maxn = 60, maxm = 15, mod = 1e9 + 7;

__int128_t bin_pow(__int128_t a, __int128_t b) {
    a %= mod, b %= mod;
    __int128_t res = 1;
    while (b) {
        if (b & 1) res = (res * a) % mod;
        a = (a * a) % mod;
        b >>= 1;
    }
    return res % mod;
}

__int128_t inv(__int128_t x) {
    return bin_pow(x, mod - 2);
}

__int128_t n, m, k;

namespace SNM {
    struct Node {
        __int128_t par;
        __int128_t sz;
    };

    Node tree[maxn];

    void init() {
        for (__int128_t i = 0; i < n; ++i) tree[i].par = i, tree[i].sz = 1;
    }

    __int128_t get_root(__int128_t v) {
        if (tree[v].par == v) return v;
        return tree[v].par = get_root(tree[v].par);
    }

    void merge(__int128_t a, __int128_t b) {
        a = get_root(a), b = get_root(b);
        if (tree[a].sz < tree[b].sz) swap(a, b);
        tree[b].par = a;
        tree[a].sz += tree[b].sz;
    }
}

vector<__int128_t> g[maxn];
__int128_t hs[maxn];
__int128_t par[maxn];
pair<__int128_t, __int128_t> paths[maxm];
bool used[maxn];

void dfs1(__int128_t v, __int128_t h, __int128_t p = -1) {
    hs[v] = h;
    par[v] = p;
    if (p != -1) g[v].erase(find(g[v].begin(), g[v].end(), p));
    for (auto u: g[v]) dfs1(u, h + 1, v);
}

__int128_t lca(__int128_t a, __int128_t b) {
    while (a != b) {
        if (hs[a] < hs[b]) swap(a, b);
        a = par[a];
    }
    return a;
}

__int128_t cnt_by_mask(__int128_t mask) {
    for (__int128_t i = 0; i < n; ++i) used[i] = false;
    SNM::init();
    __int128_t cnt = 0;
    for (__int128_t i = 0; i < m; ++i) {
        if (!(mask & (1 << i))) continue;
        __int128_t u = paths[i].first, v = paths[i].second;
        __int128_t clc = lca(u, v);
        while (!(u == clc || par[u] == clc) || !(v == clc || par[v] == clc)) {
            if (hs[u] < hs[v]) swap(u, v);
            if (!used[u]) ++cnt, used[u] = true;
            SNM::merge(u, par[u]);
            u = par[u];
        }
        if (hs[u] < hs[v]) swap(u, v);
        if (v != clc) SNM::merge(u, v);
        if (!used[u]) ++cnt, used[u] = true;
        if (v != clc && !used[v]) ++cnt, used[v] = true;
    }
    __int128_t roots_cnt = 0;
    for (__int128_t i = 0; i < n; ++i) if (SNM::tree[i].par == i && SNM::tree[i].sz > 1) ++roots_cnt;
    return bin_pow(k, roots_cnt) * bin_pow(k, n - cnt - 1) % mod;
}

void solve() {
    int _n, _m, _k;;
    cin >> _n >> _m >> _k;
    n = _n, m = _m, k = _k;
    for (__int128_t i = 1, u, v; i < n; ++i) {
        int _u, _v;
        cin >> _u >> _v, --_u, --_v;
        v = _v, u = _u;
        g[u].push_back(v), g[v].push_back(u);
    }
    for (__int128_t i = 0, a, b; i < m; ++i) {
        int _a, _b;
        cin >> _a >> _b, --_a, --_b;
        a = _a, b = _b;
        paths[i] = {a, b};
    }
    dfs1(0, 0);
    __int128_t ans = 0;
    for (__int128_t mask = 1; mask < (1 << m); ++mask) {
        __int128_t ones_cnt = 0;
        for (__int128_t t = mask; t; t >>= 1) ones_cnt += t & 1;
        if (ones_cnt & 1) ans = (ans + cnt_by_mask(mask)) % mod;
        else ans = (((ans - cnt_by_mask(mask)) % mod) + mod) % mod;
    }
    cout << static_cast<long long>(((bin_pow(k, n - 1) - ans) % mod + mod) % mod) << "\n";
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
#ifdef LOCAL
    freopen("input.txt", "r",stdin);
    freopen("output.txt", "w",stdout);
#elifdef TEST
    freopen("input.txt", "r",stdin);
#endif
    solve();
    return 0;
}
